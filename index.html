                    return sum + (libraryData[subject.code][idx]?.length || 0);

                }, 0);

                

                return `

                    <div class="subject-card" onclick="showSubject(${index})">

                        <div class="subject-icon">${subject.icon}</div>

                        <h3>${subject.name}</h3>

                        <p>${subject.code}</p>

                        <p style="margin-top: 0.5rem; color: var(--accent);">${totalFiles} resource(s) available</p>

                    </div>

                `;

            }).join('');

        }


        function showHome() {

            document.getElementById('homeView').style.display = 'block';

            document.getElementById('modulesView').style.display = 'none';

            document.getElementById('filesView').style.display = 'none';

            document.getElementById('breadcrumb').innerHTML = '<span class="active" onclick="showHome()">Home</span>';

            currentSubject = null;

            currentModule = null;

        }


        function showSubject(index) {

            currentSubject = index;

            const subject = subjects[index];

            

            document.getElementById('homeView').style.display = 'none';

            document.getElementById('modulesView').style.display = 'block';

            document.getElementById('filesView').style.display = 'none';

            

            document.getElementById('subjectTitle').textContent = subject.name;

            document.getElementById('addLinkBtn').style.display = isAdminMode ? 'block' : 'none';

            document.getElementById('breadcrumb').innerHTML = `

                <span onclick="showHome()">Home</span> 

                <span>›</span> 

                <span class="active">${subject.name}</span>

            `;

            

            const modulesList = document.getElementById('modulesList');

            modulesList.innerHTML = subject.modules.map((module, idx) => {

                const fileCount = libraryData[subject.code][idx]?.length || 0;

                return `

                    <div class="module-item" onclick="showModule(${idx})">

                        <div class="module-info">

                            <h4>${module}</h4>

                            <p>${fileCount} resource(s) available</p>

                        </div>

                        <div class="module-actions">

                            <button class="btn btn-secondary" onclick="event.stopPropagation(); showModule(${idx})">View Resources →</button>

                        </div>

                    </div>

                `;

            }).join('');

        }


        function showModules() {

            showSubject(currentSubject);

        }


        function showModule(moduleIndex) {

            currentModule = moduleIndex;

            const subject = subjects[currentSubject];

            const moduleName = subject.modules[moduleIndex];

            

            document.getElementById('modulesView').style.display = 'none';

            document.getElementById('filesView').style.display = 'block';

            

            document.getElementById('moduleTitle').textContent = moduleName;

            document.getElementById('addLinkBtn2').style.display = isAdminMode ? 'block' : 'none';

            document.getElementById('breadcrumb').innerHTML = `

                <span onclick="showHome()">Home</span> 

                <span>›</span> 

                <span onclick="showSubject(${currentSubject})">${subject.name}</span>

                <span>›</span>

                <span class="active">${moduleName}</span>

            `;

            

            renderFiles();

        }


        function renderFiles() {

            const filesList = document.getElementById('filesList');

            const moduleFiles = libraryData[subjects[currentSubject].code][currentModule] || [];

            

            if (moduleFiles.length === 0) {

                filesList.innerHTML = `

                    <div class="empty-state">

                        <h3>📚 No resources added yet</h3>

                        <p>${isAdminMode ? 'Click "Add Link" to add Google Drive resources' : 'Resources will appear here once added by admin'}</p>

                    </div>

                `;

                return;

            }

            

            filesList.innerHTML = moduleFiles.map((file, index) => `

                <div class="file-item">

                    <div class="file-info">

                        <div class="file-icon">📕</div>

                        <div>

                            <h4>${file.name}</h4>

                            <p style="color: var(--text-secondary); font-size: 0.9rem;">

                                ${file.description || 'No description'} • Added on ${file.date}

                            </p>

                        </div>

                    </div>

                    <div class="module-actions">

                        <button class="btn btn-primary" onclick="viewFile(${index})">View</button>

                        ${isAdminMode ? `<button class="btn btn-secondary" onclick="deleteFile(${index})" style="background: #c93450;">Delete</button>` : ''}

                    </div>

                </div>

            `).join('');

        }


        function showAddLinkModal() {

            if (!isAdminMode) {

                alert('Please enable Admin Mode first');

                return;

            }

            

            if (currentSubject === null || currentModule === null) {

                alert('Please select a module first');

                return;

            }

            

            document.getElementById('addLinkModal').classList.add('active');

        }


        function closeModal() {

            document.getElementById('addLinkModal').classList.remove('active');

            document.getElementById('linkFileName').value = '';

            document.getElementById('linkDescription').value = '';

            document.getElementById('driveLink').value = '';

        }


        function addDriveLink(event) {

            event.preventDefault();

            

            const fileName = document.getElementById('linkFileName').value;

            const description = document.getElementById('linkDescription').value;

            const driveLink = document.getElementById('driveLink').value;

            

            const file = {

                name: fileName,

                description: description,

                link: driveLink,

                date: new Date().toLocaleDateString()

            };

            

            libraryData[subjects[currentSubject].code][currentModule].push(file);

            saveData();

            closeModal();

            renderFiles();

            alert('✅ Link added successfully!');

        }


        function viewFile(index) {

            const file = libraryData[subjects[currentSubject].code][currentModule][index];

            

            document.getElementById('viewFileName').textContent = file.name;

            document.getElementById('viewFileDescription').textContent = file.description || 'No description provided';

            document.getElementById('viewFileDate').textContent = file.date;

            document.getElementById('viewFileLink').textContent = file.link;

            

            currentViewLink = file.link;

            

            document.getElementById('viewModal').classList.add('active');

        }


        function closeViewModal() {

            document.getElementById('viewModal').classList.remove('active');

            currentViewLink = '';

        }


        function openDriveLink() {

            if (currentViewLink) {

                window.open(currentViewLink, '_blank');

            }

        }


        function deleteFile(index) {

            if (confirm('Are you sure you want to delete this resource?')) {

                libraryData[subjects[currentSubject].code][currentModule].splice(index, 1);

                saveData();

                renderFiles();

                alert('✅ Resource deleted');

            }

        }


        function toggleTheme() {

            const root = document.documentElement;

            const button = document.querySelector('.theme-toggle');

            

            if (root.classList.contains('light')) {

                root.classList.remove('light');

                button.textContent = '🌙 Dark Mode';

            } else {

                root.classList.add('light');

                button.textContent = '☀️ Light Mode';

            }

        }
